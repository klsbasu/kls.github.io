from gturtle import *
import random

def random_fill_color():
    r = random.randint(0, 255)
    g = random.randint(0, 255)
    b = random.randint(0, 255)
    setFillColor(r, g, b)

  • Verwende Zufallsfarben, for- oder repeat-Schleifen, neue Winkel oder Formen, um deine Designs zu variieren.
