from gturtle import *
        
def onKeyPressed(key):
    if key == "ArrowUp":
        forward(100)
    elif key == "d":
        right(90)
        forward(100)                      

makeTurtle(keyPressed = onKeyPressed)

Mit makeTurtle(keyPressed=onKeyPressed)
wird die Turtle so eingestellt, dass sie auf Tastendrücke
reagiert:
  • Wenn die Pfeil-nach-oben-Taste gedrückt
    wird, bewegt sich die Turtle um 100 Pixel vorwärts.
  • Wenn die Taste d gedrückt wird, dreht sich
    die Turtle um 90 Grad nach rechts und bewegt
    sich anschließend um 100 Pixel vorwärts.

from gturtle import *
from time import sleep

posX = 0
speedX = 2

def paintBall(color):
    setPenColor(color)
    setPos(posX, 0)
    
    dot(21)

def update():
    global posX, speedX
    
    if posX >= 150:
        speedX *= -1
        
    if posX <= -150:
        speedX *= -1
        
    paintBall("white")
    posX += speedX
    paintBall("red")

makeTurtle()
hideTurtle()

repeat:
    key = getKeyCode()
    if key != 0:
        break
    update()
    sleep(0.05)
