Programmiren in der Arduino-IDE



Grundlegende Struktur
void setup() Wird einmal zu Beginn ausgeführt. 
void setup() {
  pinMode(13, OUTPUT);  // Pin 13 als Ausgang definieren (z.B. LED)
}
void loop() Wird danach endlos wiederholt. 
void loop() {
  digitalWrite(13, HIGH);  // LED einschalten
  delay(500);              // 500 ms warten
  digitalWrite(13, LOW);   // LED ausschalten
  delay(500);              // 500 ms warten
}


Datentypen
int Ganze Zahl 
int zahl = 42;  // Ganze Zahl speichern
long Große Ganzzahl 
long grosseZahl = 1234567890;  // Sehr große Zahl
float Kommazahl 
float spannung = 3.3;  // Kommazahl speichern
byte 8 Bit positive Zahl 
byte ledPin = 13;  // Wert zwischen 0 und 255
bool Wahrheitswert 
bool istAn = true;  // true oder false speichern
String Zeichenkette (Text) 
String name = "Arduino";  // Text speichern


Eingabe und Ausgabe
pinMode(Pin, Modus) Setzt Pin als INPUT oder OUTPUT. 
pinMode(13, OUTPUT);  // Pin 13 als Ausgang setzen
digitalWrite(Pin, Wert) Setzt digitalen Pin auf HIGH oder LOW. 
digitalWrite(13, HIGH);  // Pin 13 auf HIGH (z.B. LED an)
digitalRead(Pin) Liest digitalen Pin (HIGH/LOW). 
int taster = digitalRead(2);  // Zustand von Taster an Pin 2 lesen
analogRead(Pin) Liest analogen Wert (0–1023). 
int sensorWert = analogRead(A0);  // Sensorwert von Pin A0 lesen
analogWrite(Pin, Wert) Gibt PWM-Signal (0–255) aus. 
analogWrite(9, 128);  // PWM-Signal mit halber Stärke an Pin 9


Zeitfunktionen
delay(ms) Programm pausiert für bestimmte Zeit in ms. 
delay(1000);  // 1 Sekunde warten
millis() Millisekunden seit Programmstart. 
unsigned long zeit = millis();  // Zeit seit Start speichern


Bedingungen
Vergleichsoperatoren: ==, !=, >, <, >=, <=

Logische Operatoren: && (UND), || (ODER)
if(Bedingung) Führt Code aus, wenn Bedingung erfüllt ist. 
if (taster == HIGH) {
  digitalWrite(13, HIGH);  // LED einschalten, wenn Taster gedrückt
}
else Führt Code aus, wenn vorherige Bedingung falsch war. 
else {
  digitalWrite(13, LOW);  // LED ausschalten, wenn Taster nicht gedrückt
}
else if(Bedingung) Weitere Bedingungen prüfen. 
else if (zahl > 10) {
  Serial.println("Groß");  // Ausgabe, wenn Bedingung zutrifft
}


Schleifen
for(Initialisierung; Bedingung; Aktualisierung) Wiederholt Anweisungen eine feste Anzahl Mal. 
for (int i = 0; i < 10; i++) {
  // 10-mal etwas tun
}
while(Bedingung) Wiederholt solange Bedingung erfüllt ist. 
while (x < 5) {
  // solange x kleiner als 5 ist
}


Serielle Kommunikation
Serial.begin(baudrate) Startet serielle Kommunikation. 
Serial.begin(9600);  // Kommunikation mit 9600 Baud starten
Serial.print(Daten) Daten ausgeben ohne Zeilenumbruch. 
Serial.print("Hallo");  // Gibt "Hallo" aus (kein Umbruch)
Serial.println(Daten) Daten ausgeben mit Zeilenumbruch. 
Serial.println("Welt");  // Gibt "Welt" mit Zeilenumbruch aus


Eigene Funktionen
Datentyp Funktionsname(Parameter) Definiert eine eigene Funktion. 
int addiere(int a, int b) {
  return a + b;  // Gibt die Summe von a und b zurück
}


Zufallszahlen
randomSeed(Wert) Initialisiert den Zufallsgenerator. 
randomSeed(analogRead(A0));  // Startwert für Zufallszahlen setzen
random(Max) Gibt Zufallszahl 0 bis Max-1. 
random(10);  // Zufallszahl von 0 bis 9
random(Min, Max) Gibt Zufallszahl zwischen Min und Max-1. 
random(5, 15);  // Zufallszahl von 5 bis 14


Tonausgabe
tone(Pin, Frequenz) Spielt einen Ton. 
tone(8, 440);  // Spielt Ton mit 440 Hz
tone(Pin, Frequenz, Dauer) Spielt Ton für bestimmte Dauer. 
tone(8, 440, 500);  // Ton für 500 ms abspielen
noTone(Pin) Stoppt Ton auf Pin. 
noTone(8);  // Ton auf Pin 8 stoppen


LCD-Display Steuerung
Bibliothek #include <SerLCD.h> erforderlich.
lcd.setPosition(Zeile, Spalte) Setzt den Cursor. 
lcd.setPosition(1, 0);  // Cursor in Zeile 1, Spalte 0 setzen
lcd.print(Text) Text auf LCD ausgeben. 
lcd.print("Hallo");  // Gibt "Hallo" auf dem Display aus
lcd.clear() Display löschen. 
lcd.clear();  // Löscht den LCD-Inhalt
